    <div class="text-gray-800 dark:text-gray-200">

<script is:inline>{`
  (function () {
    var lb = document.getElementById("gd-lightbox");
    if (!lb || lb.dataset.gdInit) return;
    lb.dataset.gdInit = "1";
    document.body.appendChild(lb);
    var img = lb.querySelector(".gd-lightbox-img");
    var thumbs = Array.prototype.slice.call(document.querySelectorAll(".gd-thumb"));
    var srcs = thumbs.map(function (t) { return t.getAttribute("data-gd-src"); });
    var idx = 0;
    function show(i) {
      idx = (i + srcs.length) % srcs.length;
      img.src = srcs[idx];
      lb.classList.add("gd-open");
      document.body.style.overflow = "hidden";
    }
    function close() {
      lb.classList.remove("gd-open");
      document.body.style.overflow = "";
    }
    thumbs.forEach(function (t, i) {
      t.addEventListener("click", function () { show(i); });
    });
    lb.addEventListener("click", function (e) {
      if (e.target === lb || e.target.classList.contains("gd-lightbox-close")) close();
    });
    lb.querySelector(".gd-prev").addEventListener("click", function (e) { e.stopPropagation(); show(idx - 1); });
    lb.querySelector(".gd-next").addEventListener("click", function (e) { e.stopPropagation(); show(idx + 1); });
    document.addEventListener("keydown", function (e) {
      if (!lb.classList.contains("gd-open")) return;
      if (e.key === "Escape") close();
      else if (e.key === "ArrowLeft") show(idx - 1);
      else if (e.key === "ArrowRight") show(idx + 1);
    });
  })();
`}</script>
